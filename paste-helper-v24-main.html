// OnGameTick -- Called every game tick (~600ms). Core dispatch loop.
// =============================================================================
function OnGameTick() {
    tickCounter++;
    ticksSinceLastBreak++;
    ticksSinceAntifire++;
    ticksSinceAntipoison++;
    ticksSinceCombatPot++;

    if (isExecuting) {
        stateTimeout++;
        if (stateTimeout > 30) {
            isExecuting = false;
            stateTimeout = 0;
            Game.sendGameMessage("State timeout - resetting execution lock", "SlayerBot");
        }
        return;
    }

    // ================================================================
    // SCHEDULE CHECK -- Are we within play hours?
    // ================================================================
    if (config.enableSchedule.getValue()) {
        var schedStart = config.startHour.getValue();
        var schedEnd = config.endHour.getValue();
        if (!isWithinSchedule(schedStart, schedEnd)) {
            if (state !== STATES.OUTSIDE_SCHEDULE && state !== STATES.TAKING_BREAK) {
                stateBeforeBreak = state;
                state = STATES.OUTSIDE_SCHEDULE;
                outsideScheduleLogout = false;
                Game.sendGameMessage("[Schedule] Outside play hours -- stopping", "Bot");
                config.scheduleOverlay.setValue("Schedule: OUTSIDE HOURS");
            }
        } else {
            config.scheduleOverlay.setValue("Schedule: Active");
        }
    }

    // ================================================================
    // BREAK CHECK -- Time for a random human break?
    // ================================================================
    if (config.enableBreaks.getValue() && state !== STATES.TAKING_BREAK && state !== STATES.OUTSIDE_SCHEDULE) {
        // Initialize first break timer on session start
        if (!sessionStarted) {
            sessionStarted = true;
            ticksSinceLastBreak = 0;
            var initMin = config.minPlayMinutes.getValue();
            var initMax = config.maxPlayMinutes.getValue();
            nextBreakAtTick = calculateNextBreakDelay(initMin, initMax);
            config.breakOverlay.setValue("Break: in ~" + ticksToMinutes(nextBreakAtTick) + "m");
        }

        // Check if it is break time
        if (ticksSinceLastBreak >= nextBreakAtTick) {
            // Only break when safe -- not mid-combat or death
            var safeToBreak = (state !== STATES.FIGHT_TASK && state !== STATES.SAFESPOT_FIGHT &&
                state !== STATES.HANDLE_DEATH && state !== STATES.RECOVER_GRAVE &&
                state !== STATES.EAT_FOOD && state !== STATES.DRINK_PRAYER);

            if (safeToBreak) {
                stateBeforeBreak = state;
                state = STATES.TAKING_BREAK;
                breakReason = getRandomBreakReason();
                var bMin = config.minBreakMinutes.getValue();
                var bMax = config.maxBreakMinutes.getValue();
                var duration = calculateBreakDuration(bMin, bMax);
                breakEndTick = tickCounter + duration;
                lastBreakDurationMins = ticksToMinutes(duration);
                totalBreaksTaken++;
                Game.sendGameMessage("[Break] " + breakReason + " (~" + lastBreakDurationMins + " min)", "Bot");
                config.breakOverlay.setValue("Break: " + breakReason + " (" + lastBreakDurationMins + "m)");
                return;
            }
        } else {
            // Update countdown overlay
            var remaining = nextBreakAtTick - ticksSinceLastBreak;
            if (tickCounter % 100 === 0) {
                config.breakOverlay.setValue("Break: in ~" + ticksToMinutes(remaining) + "m");
            }
        }
    }

    // Update cached stats every 10 ticks
    if (tickCounter % 10 === 0) {
        updateCachedStats();
    }

    // Update overlay
    updateOverlay();

    // Track state changes
    if (state !== lastState) {
        stateTickCounter = 0;
        lastState = state;
        Game.sendGameMessage("State -> " + state, "SlayerBot");
    }
    stateTickCounter++;

    // Stuck detection: if we are in a non-idle state for too long, reset
    if (stateTickCounter > 500 && state !== STATES.IDLE && state !== STATES.FIGHT_TASK && state !== STATES.SAFESPOT_FIGHT) {
        Game.sendGameMessage("Stuck in " + state + " for 500 ticks, resetting to CHECK_STATS", "SlayerBot");
        state = STATES.CHECK_STATS;
        stateTickCounter = 0;
        retryCount = 0;
    }

    // Death detection: check if we are at Lumbridge spawn
    if (state !== STATES.HANDLE_DEATH && state !== STATES.RECOVER_GRAVE && state !== STATES.STARTUP) {
        if (isAtDeathSpawn()) {
            if (state === STATES.FIGHT_TASK || state === STATES.SAFESPOT_FIGHT || state === STATES.WALK_TO_TASK) {
                Game.sendGameMessage("Death detected! Saving death location.", "SlayerBot");
                state = STATES.HANDLE_DEATH;
                return;
            }
        }
    }

    // Level up detection
    if (levelUpDetected) {
        levelUpDetected = false;
        updateCachedStats();
    }

    // =========================================================================

    // =========================================================================
    ticksSinceRepairCheck++;
    if (ticksSinceRepairCheck >= repairCheckInterval && state !== STATES.REPAIR_BARROWS &&
        state !== STATES.TAKING_BREAK && state !== STATES.OUTSIDE_SCHEDULE &&

function drinkPotionFromGroup(potGroup) {
    for (var i = 0; i < potGroup.length; i++) {
        if (Game.info.inventory.hasItem(potGroup[i], 1)) {
            isExecuting = true;
            var potId = potGroup[i];
            Utility.invokeLater(function() {
                Game.interact.inventory.consumeItem(potId);
                isExecuting = false;
            }, Utility.getDelay());
            return true;
        }
    }
    return false;
}

        var item = groundItems[i];
        var itemId = item.getId();

        // Distance check: only pick up items within 5 tiles
        var itemLoc = item.getWorldLocation();
        var dx = Math.abs(itemLoc.getX() - playerLoc.x);
        var dy = Math.abs(itemLoc.getY() - playerLoc.y);
        if (dx > 5 || dy > 5) continue;

        // Always pick up: bones, clue scrolls
        var isBone = ALL_BONE_IDS.indexOf(itemId) !== -1;
        var isClue = CLUE_SCROLLS.indexOf(itemId) !== -1;

        if (isBone && config.buryBones.getValue()) {

            Utility.invokeLater(function() {
                Game.interact.groundItem.pickup(item);
                isExecuting = false;
            }, Utility.getDelay());

            isExecuting = true;
            Utility.invokeLater(function() {
                Game.interact.groundItem.pickup(item);
                isExecuting = false;
            }, Utility.getDelay());
            return true;
        }

        // Pick up stackable/notable items
        isExecuting = true;
        Utility.invokeLater(function() {
            Game.interact.groundItem.pickup(item);
            isExecuting = false;
        }, Utility.getDelay());

function handleSurvivalChecks() {
    var currentHP = Client.getBoostedSkillLevels(Skill.HITPOINTS);
    var maxHP = Client.getRealSkillLevels(Skill.HITPOINTS);

    // Emergency eat
    if (currentHP <= Math.floor(maxHP * 0.4)) {
        eatFood();
        return;
    }

    // Prayer check
    if (config.usePrayer.getValue()) {
        var currentPrayer = Client.getBoostedSkillLevels(Skill.PRAYER);
        if (currentPrayer <= 10) {
            drinkPrayerPot();
            return;
        }
    }

    // Dragon task antifire
    if (currentTaskMonster && isDragonTask(currentTaskMonster.name)) {
        if (ticksSinceAntifire > 550 || ticksSinceAntifire === 0) {
            drinkAntifire();
        }
    }

    // Poison monsters antipoison
    if (currentTaskMonster && canPoison(currentTaskMonster.name)) {
        if (ticksSinceAntipoison > 800 || ticksSinceAntipoison === 0) {
            drinkAntipoison();
        }
    }

    // Combat potion re-dose
    if (ticksSinceCombatPot > 50) {
        var boostedAtk = Client.getBoostedSkillLevels(Skill.ATTACK);
        var realAtk = Client.getRealSkillLevels(Skill.ATTACK);
        if (boostedAtk <= realAtk + 2) {
            if (drinkCombatPotion()) {
                ticksSinceCombatPot = 0;
            }
        }
        var boostedStr = Client.getBoostedSkillLevels(Skill.STRENGTH);
        var realStr = Client.getRealSkillLevels(Skill.STRENGTH);
        if (boostedStr <= realStr + 2) {
            if (drinkStrengthPotion()) {
                ticksSinceCombatPot = 0;
            }
        }
    }
}

function handleTalkToMaster() {
    if (!selectedMaster) {
        state = STATES.SELECT_MASTER;
        return;
    }

    // Find the master NPC
    var masterNpc = Game.info.npc.getNearest([selectedMaster.npcId]);
    if (!masterNpc) {
        // Not close enough, keep walking
        if (retryCount < maxRetries) {
            retryCount++;
            state = STATES.WALK_TO_MASTER;
        } else {
            Game.sendGameMessage("Cannot find master NPC after retries, resetting.", "SlayerBot");
            retryCount = 0;
            state = STATES.SELECT_MASTER;
        }
        return;
    }

    taskReceived = false;
    isExecuting = true;
    Utility.invokeLater(function() {
        Game.interact.npc.action(masterNpc, MenuAction.NPC_SECOND_OPTION);
        isExecuting = false;
    }, Utility.getDelay());

    state = STATES.WAIT_FOR_TASK;
    waitTicks = 0;
}

// ---- WAIT_FOR_TASK ----
function handleWaitForTask() {
    waitTicks++;

    if (taskReceived) {
        state = STATES.READ_TASK;
        waitTicks = 0;
        return;
    }

    // Handle dialog continuation
    try {
        var dialogWidget = Game.info.widget.getWidget(231, 4);
        if (dialogWidget) {
            isExecuting = true;
            Utility.invokeLater(function() {
                Game.interact.widget.interact(dialogWidget, MenuAction.NPC_FIRST_OPTION);
                isExecuting = false;
            }, Utility.getDelay());
            return;
        }
        var continueWidget = Game.info.widget.getWidget(217, 4);
        if (continueWidget) {
            isExecuting = true;
            Utility.invokeLater(function() {
                Game.interact.widget.interact(continueWidget, MenuAction.NPC_FIRST_OPTION);
                isExecuting = false;
            }, Utility.getDelay());
            return;
        }
    } catch (e) {}

    if (waitTicks > 30) {
        Game.sendGameMessage("Task not received after 30 ticks, retrying talk.", "SlayerBot");
        waitTicks = 0;
        retryCount++;
        if (retryCount > maxRetries) {
            retryCount = 0;
            state = STATES.SELECT_MASTER;
        } else {
            state = STATES.TALK_TO_MASTER;
        }
    }
}

// ---- READ_TASK ----
function handleReadTask() {
    if (!currentTaskName || killsLeft <= 0) {
        Game.sendGameMessage("No valid task found, returning to master.", "SlayerBot");
        state = STATES.SELECT_MASTER;
        return;
    }

    Game.sendGameMessage("Task: Kill " + killsLeft + " " + currentTaskName, "SlayerBot");

    // Look up monster data
    currentTaskMonster = findMonster(currentTaskName);
    if (!currentTaskMonster) {
        Game.sendGameMessage("WARNING: Unknown monster '" + currentTaskName + "', returning to master.", "SlayerBot");
        state = STATES.SELECT_MASTER;
        return;
    }

    // v24: Equip task bracelet (expeditious or slaughter)
    equipTaskBracelet(currentTaskName);

    // v24: Check if safespot is available and configured
    var safespotData = null;
    if (config.useSafespots.getValue()) {
        safespotData = hasSafeSpot(currentTaskName);
    }
    if (safespotData) {
        usingSafespot = true;
        Game.sendGameMessage("Safespot available for " + currentTaskName + "!", "SlayerBot");
    } else {
        usingSafespot = false;
    }

    // Move to supply check
    state = STATES.CHECK_SUPPLIES;
}

// ---- CHECK_SUPPLIES ----
function handleCheckSupplies() {
    missingItems = [];
    needsRestock = false;

    // Check food
    var foodCount = countFood();
    var desiredFood = getDesiredFoodCount();
    if (foodCount < desiredFood) {
        missingItems.push({type: "food", current: foodCount, needed: desiredFood});
        needsRestock = true;
    }

    // Check prayer pots
    if (config.usePrayer.getValue()) {
        var ppotCount = countPrayerPots();
        var desiredPpots = getDesiredPrayerPotCount();
        if (ppotCount < desiredPpots) {
            missingItems.push({type: "prayer_pot", current: ppotCount, needed: desiredPpots});
            needsRestock = true;
        }
    }

    // v24: Check arrows (ranged users or safespot with ranged)
    if (config.useRanged.getValue() || usingSafespot) {
        var arrowCount = countArrows();
        if (arrowCount < 100) {
            missingItems.push({type: "arrows", current: arrowCount, needed: 500});
            needsRestock = true;
        }
    }

    // v24: Check runes (magic users)
    if (config.useMagic.getValue()) {
        var runeCount = countRunes();
        if (runeCount < 100) {
            missingItems.push({type: "runes", current: runeCount, needed: 500});
            needsRestock = true;
        }
    }

    // v24: Check antifire potions (dragon tasks)
    if (currentTaskMonster && isDragonTask(currentTaskMonster.name)) {
        var hasAntifire = hasPotionInGroup(SUPER_ANTIFIRE_POTS) ||
                          hasPotionInGroup(EXTENDED_ANTIFIRE_POTS) ||
                          hasPotionInGroup(ANTIFIRE_POTS);
        if (!hasAntifire) {
            missingItems.push({type: "antifire", current: 0, needed: 2});
            needsRestock = true;
        }
    }

    // v24: Check antipoison (poisoning tasks)
    if (currentTaskMonster && canPoison(currentTaskMonster.name)) {
        var hasAntipois = hasPotionInGroup(ANTIVENOM_PLUS_IDS) ||
                          hasPotionInGroup(ANTIVENOM_IDS) ||
                          hasPotionInGroup(SUPER_ANTIPOISON_IDS);
        if (!hasAntipois) {
            missingItems.push({type: "antipoison", current: 0, needed: 1});
            needsRestock = true;
        }
    }

    // v24: Check combat potions
    var hasCombatPot = hasPotionInGroup(SUPER_COMBAT_POTS) ||
                       hasPotionInGroup(SUPER_ATTACK_POTS) ||
                       hasPotionInGroup(ATTACK_POTS);
    if (!hasCombatPot) {
        missingItems.push({type: "combat_pot", current: 0, needed: 1});
        needsRestock = true;
    }

    // v24: Check cannonballs (only for multicombat tasks)
    if (config.useCannon.getValue() && shouldUseCannon(currentTaskMonster, true)) {
        var cballs = Game.info.inventory.getItemCount(CANNON.CANNONBALL);
        if (cballs < 200) {
            missingItems.push({type: "cannonballs", current: cballs, needed: CANNON.MIN_CANNONBALLS});
            needsRestock = true;
        }
    }

    // v24: Check Barrows gear condition -- repair at 25% before it breaks!
    var barrowsRepairNeeded = checkEquippedBarrowsCondition();
    if (barrowsRepairNeeded.length > 0) {
        for (var br = 0; br < barrowsRepairNeeded.length; br++) {
            Game.sendGameMessage("[SlayerBot] WARNING: " + barrowsRepairNeeded[br].piece +
                " is at " + barrowsRepairNeeded[br].condition + "% - repair needed!", "Bot");
        }
        // Save current state and go repair
        stateBeforeRepair = state;
        repairPhase = "walk_to_bob";
        state = STATES.REPAIR_BARROWS;
        return;
    }

    // Check special finish items
    if (config.specialFinish.getValue() && currentTaskMonster) {
        var monName = currentTaskMonster.name.toLowerCase();
        if (monName.indexOf("gargoyle") !== -1 && !Game.info.inventory.hasItem(ROCK_HAMMER_ID, 1)) {
            missingItems.push({type: "special", itemId: ROCK_HAMMER_ID, name: "rock_hammer"});
            needsRestock = true;
        }
        if (monName.indexOf("rockslug") !== -1 && !Game.info.inventory.hasItem(ICE_COOLER_ID, 1)) {
            missingItems.push({type: "special", itemId: ICE_COOLER_ID, name: "ice_cooler"});
            needsRestock = true;
        }
        if (monName.indexOf("lizard") !== -1 && !Game.info.inventory.hasItem(ICE_COOLER_ID, 1)) {
            missingItems.push({type: "special", itemId: ICE_COOLER_ID, name: "ice_cooler"});
            needsRestock = true;
        }
    }

    if (needsRestock) {
        Game.sendGameMessage("Need restock: " + missingItems.length + " item type(s) missing", "SlayerBot");
        state = STATES.WALK_TO_BANK;
    } else {
        Game.sendGameMessage("Supplies OK, heading to task.", "SlayerBot");
        state = STATES.WALK_TO_TASK;
    }
}

// ---- BANK_SUPPLIES ----
function handleBankSupplies() {
    // Open bank if not open
    if (!Game.info.bank.isOpen()) {
        isExecuting = true;
        Utility.invokeLater(function() {
            Game.interact.bank.openNearest();
            isExecuting = false;
        }, Utility.getDelay());
        return;
    }

    // Deposit all first
    isExecuting = true;
    Utility.invokeLater(function() {
        Game.interact.bank.depositAllInventory();
        isExecuting = false;
    }, Utility.getDelay());

    // Wait a tick then withdraw
    waitTicks++;
    if (waitTicks < 3) return;
    waitTicks = 0;

    var withdrawDone = true;

    for (var i = 0; i < missingItems.length; i++) {
        var item = missingItems[i];

        if (item.type === "food") {
            var foodId = 379;
            try { foodId = config.foodType.getValue(); } catch(e) {}
            withdrawBankItem(foodId, item.needed);
        } else if (item.type === "prayer_pot") {
            withdrawPotionFromBank(PRAYER_POTS, item.needed);
        } else if (item.type === "arrows") {
            var bestArrow = getBestArrows(rangedLevel);
            if (bestArrow) {
                withdrawBankItem(bestArrow.id, item.needed);
            } else {
                // Need to craft arrows
                if (config.selfSufficient.getValue()) {
                    stateBeforeInterrupt = STATES.CHECK_SUPPLIES;
                    state = STATES.CRAFT_ARROWS;
                    Game.interact.bank.close();
                    return;
                }
            }
        } else if (item.type === "runes") {
            // Try to withdraw runes from bank
            var spell = getBestSpell(magicLevel);
            if (spell) {
                var runes = getRunesForSpell(spell.key);
                if (runes) {
                    var runeKeys = Object.keys(runes);
                    for (var r = 0; r < runeKeys.length; r++) {
                        var runeId = parseInt(runeKeys[r]);
                        var runeQty = runes[runeKeys[r]] * 200;
                        withdrawBankItem(runeId, runeQty);
                    }
                }
            }
            // If we don't have enough, craft runes
            if (config.selfSufficient.getValue() && countRunes() < 50) {
                stateBeforeInterrupt = STATES.CHECK_SUPPLIES;
                state = STATES.CRAFT_RUNES;
                Game.interact.bank.close();
                return;
            }
        } else if (item.type === "antifire") {
            if (!withdrawPotionFromBank(SUPER_ANTIFIRE_POTS, item.needed)) {
                if (!withdrawPotionFromBank(EXTENDED_ANTIFIRE_POTS, item.needed)) {
                    if (!withdrawPotionFromBank(ANTIFIRE_POTS, item.needed)) {
                        if (config.selfSufficient.getValue()) {
                            stateBeforeInterrupt = STATES.CHECK_SUPPLIES;
                            brewTarget = "antifire";
                            state = STATES.BREW_POTIONS;
                            Game.interact.bank.close();
                            return;
                        }
                    }
                }
            }
        } else if (item.type === "antipoison") {
            if (!withdrawPotionFromBank(SUPER_ANTIPOISON_IDS, item.needed)) {
                if (config.selfSufficient.getValue()) {
                    stateBeforeInterrupt = STATES.CHECK_SUPPLIES;
                    brewTarget = "superantipoison";
                    state = STATES.BREW_POTIONS;
                    Game.interact.bank.close();
                    return;
                }
            }
        } else if (item.type === "combat_pot") {
            if (!withdrawPotionFromBank(SUPER_COMBAT_POTS, 1)) {
                if (!withdrawPotionFromBank(SUPER_ATTACK_POTS, 1)) {
                    if (!withdrawPotionFromBank(ATTACK_POTS, 1)) {
                        if (config.selfSufficient.getValue()) {
                            stateBeforeInterrupt = STATES.CHECK_SUPPLIES;
                            brewTarget = "supercombat";
                            state = STATES.BREW_POTIONS;
                            Game.interact.bank.close();
                            return;
                        }
                    }
                }
            }
        } else if (item.type === "cannonballs") {
            // Try to withdraw from bank first
            var bankCballs = withdrawBankItem(CANNON.CANNONBALL, item.needed);
            // If bank has fewer than needed, enter cannonball gathering state
            if (Game.info.inventory.getItemCount(CANNON.CANNONBALL) < 200) {
                Game.sendGameMessage("[SlayerBot] Low cannonballs in bank -- going to smith more", "Bot");
                state = STATES.GATHER_CANNONBALLS;
                return;
            }
        } else if (item.type === "special") {
            withdrawBankItem(item.itemId, 1);
        }
    }

    // Close bank and re-check
    isExecuting = true;
    Utility.invokeLater(function() {
        Game.interact.bank.close();
        isExecuting = false;
    }, Utility.getDelay());

    missingItems = [];
    state = STATES.CHECK_SUPPLIES;
}

function withdrawBankItem(itemId, qty) {
    try {
        // Use appropriate withdraw type based on quantity
        var wType = BankWithdrawType.WITHDRAW_ALL;
        if (qty === 1) { wType = BankWithdrawType.WITHDRAW_ONE; }
        else if (qty === 5) { wType = BankWithdrawType.WITHDRAW_FIVE; }
        else if (qty === 10) { wType = BankWithdrawType.WITHDRAW_TEN; }
        Game.interact.bank.withdrawItem(itemId, wType, false);
    } catch(e) {}
}

// ---- FIGHT_TASK (major overhaul for v24) ----
function handleFightTask() {
    if (!currentTaskMonster) {
        state = STATES.SELECT_MASTER;
        return;
    }

    // Task complete check
    if (killsLeft <= 0) {
        Game.sendGameMessage("Task complete! Killed all " + currentTaskName, "SlayerBot");
        state = STATES.TASK_COMPLETE;
        return;
    }

    var currentHP = Client.getBoostedSkillLevels(Skill.HITPOINTS);
    var maxHP = Client.getRealSkillLevels(Skill.HITPOINTS);

    // === PRIORITY 1: Death check ===
    if (currentHP <= 0) {
        saveDeathLocation();
        state = STATES.HANDLE_DEATH;
        return;
    }

    // === PRIORITY 2: Emergency teleport if critically low ===
    if (currentHP <= EMERGENCY_HP_THRESHOLD && countFood() === 0) {
        Game.sendGameMessage("CRITICAL HP with no food! Emergency retreat!", "SlayerBot");
        // Try to eat anything
        if (!eatFood()) {
            // Walk away from combat
            state = STATES.WALK_TO_BANK;
            return;
        }
    }

    // === PRIORITY 3: Eat food ===
    if (currentHP <= Math.floor(maxHP * 0.45)) {
        if (countFood() > 0) {
            eatFood();
            return;
        } else {
            // Out of food -- bank
            Game.sendGameMessage("Out of food, banking.", "SlayerBot");
            state = STATES.WALK_TO_BANK;
            return;
        }
    }

    // === PRIORITY 4: Drink prayer pot ===
    if (config.usePrayer.getValue()) {
        var currentPrayer = Client.getBoostedSkillLevels(Skill.PRAYER);
        if (currentPrayer <= 15) {
            if (countPrayerPots() > 0) {
                drinkPrayerPot();
                return;
            }
        }
    }

    // === PRIORITY 5: Anti-fire check (dragon tasks) ===
    if (currentTaskMonster && isDragonTask(currentTaskMonster.name)) {
        if (ticksSinceAntifire > 550 || ticksSinceAntifire === 0) {
            if (drinkAntifire()) {
                return;
            }
        }
    }

    // === PRIORITY 6: Anti-poison check ===
    if (currentTaskMonster && canPoison(currentTaskMonster.name)) {
        if (ticksSinceAntipoison > 800 || ticksSinceAntipoison === 0) {
            if (drinkAntipoison()) {
                return;
            }
        }
    }

    // === PRIORITY 7: Combat potion re-dose ===
    if (ticksSinceCombatPot > 50) {
        var boostedAtk = Client.getBoostedSkillLevels(Skill.ATTACK);
        var realAtk = Client.getRealSkillLevels(Skill.ATTACK);
        if (boostedAtk <= realAtk + 2) {
            if (drinkCombatPotion()) {
                ticksSinceCombatPot = 0;
                return;
            }
        }
        var boostedStr = Client.getBoostedSkillLevels(Skill.STRENGTH);
        var realStr = Client.getRealSkillLevels(Skill.STRENGTH);
        if (boostedStr <= realStr + 2) {
            if (drinkStrengthPotion()) {
                ticksSinceCombatPot = 0;
                return;
            }
        }
    }

    // === PRIORITY 8: Special attack ===
    if (config.useSpecialAttack.getValue()) {
        var specPercent = 0;
        try { specPercent = Client.getSpecialAttackPercent(); } catch(e) {}
        if (specPercent >= 25) {
            useSpecialAttack();
        }
    }

    // === PRIORITY 9: Superior monster check ===
    var superior = checkForSuperior();
    if (superior) {
        Game.sendGameMessage("SUPERIOR FOUND! Attacking superior.", "SlayerBot");
        isExecuting = true;
        Utility.invokeLater(function() {
            Game.interact.npc.attack(superior);
            isExecuting = false;
        }, Utility.getDelay());
        return;
    }

    // === PRIORITY 10: Special finish items ===
    if (config.specialFinish.getValue()) {
        var finished = handleSpecialFinishCheck();
        if (finished) return;
    }

    // === PRIORITY 11: Loot check ===
    if (!Client.getLocalPlayer().isInteracting()) {
        if (pickUpLoot()) return;
    }

    // === PRIORITY 12: Clue scroll banking ===
    if (Game.info.inventory.isFull() && hasClueScroll()) {
        Game.sendGameMessage("Inventory full with clue scroll, banking.", "SlayerBot");
        state = STATES.WALK_TO_BANK;
        return;
    }

    // === PRIORITY 13: Bone burying ===
    if (config.buryBones.getValue() && !Client.getLocalPlayer().isInteracting()) {
        for (var b = 0; b < ALL_BONE_IDS.length; b++) {
            if (Game.info.inventory.hasItem(ALL_BONE_IDS[b], 1)) {
                var boneId = ALL_BONE_IDS[b];
                isExecuting = true;
                Utility.invokeLater(function() {
                    Game.interact.inventory.consumeItem(boneId);
                    isExecuting = false;
                }, Utility.getDelay());
                return;
            }
        }
    }

    // === PRIORITY 14: Cannon reload ===
    if (config.useCannon.getValue() && cannonPlaced) {
        reloadCannon();
    }

    // === PRIORITY 15: Attack target ===
    if (!Client.getLocalPlayer().isInteracting()) {
        var npcIds = currentTaskMonster.ids;
        if (!npcIds || npcIds.length === 0) {
            Game.sendGameMessage("No NPC IDs for " + currentTaskName, "SlayerBot");
            state = STATES.SELECT_MASTER;
            return;
        }

        var target = Game.info.npc.getNearest(npcIds);
        if (target && !target.isDead()) {
            isExecuting = true;
            Utility.invokeLater(function() {
                Game.interact.npc.attack(target);
                isExecuting = false;
            }, Utility.getDelay());
        } else {
            // No monster found -- might need to wait or move
            if (stateTickCounter > 20 && !target) {
                // Try walking closer to task center
                if (!PlayerHelper.isWebWalking()) {
                    var cx = currentTaskMonster.x;
                    var cy = currentTaskMonster.y;
                    var cp = currentTaskMonster.plane || 0;
                    isExecuting = true;
                    Utility.invokeLater(function() {
                        PlayerHelper.webWalkTo(new WorldPoint(cx, cy, cp));
                        isExecuting = false;
                    }, Utility.getDelay());
                }
            }
        }
    }
}

}

// ---- SAFESPOT_FIGHT ----
function handleSafespotFight() {
    if (!currentTaskMonster) {
        state = STATES.SELECT_MASTER;
        return;
    }

    if (killsLeft <= 0) {
        Game.sendGameMessage("Safespot task complete! Killed all " + currentTaskName, "SlayerBot");
        state = STATES.TASK_COMPLETE;
        return;
    }

    var spot = hasSafeSpot(currentTaskMonster.name);
    if (!spot) {
        // Fallback to normal melee fight
        state = STATES.FIGHT_TASK;
        return;
    }

    // Move to safe tile if not there
    if (!isNear(spot.safeX, spot.safeY, 2)) {
        if (!PlayerHelper.isWebWalking()) {
            isExecuting = true;
            var sx = spot.safeX;
            var sy = spot.safeY;
            var sp = spot.safePlane;
            Utility.invokeLater(function() {
                PlayerHelper.webWalkTo(new WorldPoint(sx, sy, sp));
                isExecuting = false;
            }, Utility.getDelay());
        }
        return;
    }

    // Activate prayer if needed
    if (spot.prayAgainst) {
        activateSpecificPrayer(spot.prayAgainst);
    }

    // Survival checks
    handleSurvivalChecks();

    // Check for superior (even in safespot)
    var superiorNpc = checkForSuperior();
    if (superiorNpc) {
        Game.sendGameMessage("SUPERIOR in safespot! Attacking.", "SlayerBot");
        isExecuting = true;
        Utility.invokeLater(function() {
            Game.interact.npc.attack(superiorNpc);
            isExecuting = false;
        }, Utility.getDelay());
        return;
    }

    // Loot check
    if (!Client.getLocalPlayer().isInteracting()) {
        if (pickUpLoot()) return;
    }

    // Bone burying
    if (config.buryBones.getValue() && !Client.getLocalPlayer().isInteracting()) {
        for (var b = 0; b < ALL_BONE_IDS.length; b++) {
            if (Game.info.inventory.hasItem(ALL_BONE_IDS[b], 1)) {
                var safeBoneId = ALL_BONE_IDS[b];
                isExecuting = true;
                Utility.invokeLater(function() {
                    Game.interact.inventory.consumeItem(safeBoneId);
                    isExecuting = false;
                }, Utility.getDelay());
                return;
            }
        }
    }

    // Supply interrupts
    var fCount = countFood();
    if (fCount <= 0) {
        Game.sendGameMessage("Out of food in safespot, banking.", "SlayerBot");
        state = STATES.WALK_TO_BANK;
        return;
    }

    // Arrow/rune supply interrupt
    if (config.useRanged.getValue() && countArrows() < 10) {
        Game.sendGameMessage("Low on arrows, restocking.", "SlayerBot");
        if (config.selfSufficient.getValue()) {
            stateBeforeInterrupt = STATES.CHECK_SUPPLIES;
            state = STATES.CRAFT_ARROWS;
        } else {
            state = STATES.WALK_TO_BANK;
        }
        return;
    }
    if (config.useMagic.getValue() && countRunes() < 10) {
        Game.sendGameMessage("Low on runes, restocking.", "SlayerBot");
        if (config.selfSufficient.getValue()) {
            stateBeforeInterrupt = STATES.CHECK_SUPPLIES;
            state = STATES.CRAFT_RUNES;
        } else {
            state = STATES.WALK_TO_BANK;
        }
        return;
    }

    // Attack with ranged/magic from safe spot
    if (!Client.getLocalPlayer().isInteracting()) {
        attackWithRangedOrMagic();
    }
}

// ---- RECOVER_GRAVE ----
function handleRecoverGrave() {
    // After respawn in Lumbridge:
    // 1. Bank and withdraw emergency food
    // 2. Walk to death location
    // 3. Collect items
    // 4. Re-equip gear
    // 5. Return to task

    if (!deathLocation) {
        Game.sendGameMessage("No death location saved, re-checking supplies.", "SlayerBot");
        deathDetected = false;
        state = STATES.CHECK_SUPPLIES;
        return;
    }

    // Phase 1: Get emergency food from bank
    var foodCt = countFood();
    if (foodCt < 3) {
        // Need to bank first
        if (!Game.info.bank.isOpen()) {
            var bank = findNearestBank();
            if (bank && isNear(bank.x, bank.y, 8)) {
                isExecuting = true;
                Utility.invokeLater(function() {
                    Game.interact.bank.openNearest();
                    isExecuting = false;
                }, Utility.getDelay());
                return;
            } else if (bank) {
                if (!PlayerHelper.isWebWalking()) {
                    isExecuting = true;
                    var rbx = bank.x;
                    var rby = bank.y;
                    var rbp = bank.plane || 0;
                    Utility.invokeLater(function() {
                        PlayerHelper.webWalkTo(new WorldPoint(rbx, rby, rbp));
                        isExecuting = false;
                    }, Utility.getDelay());
                }
                return;
            }
        } else {
            // Bank is open, withdraw food
            var emergFoodId = 379;
            try { emergFoodId = config.foodType.getValue(); } catch(e) {}
            withdrawBankItem(emergFoodId, 5);
            isExecuting = true;
            Utility.invokeLater(function() {
                Game.interact.bank.close();
                isExecuting = false;
            }, Utility.getDelay());
            return;
        }
    }

    // Phase 2: Walk to death pile
    if (!isNear(deathLocation.x, deathLocation.y, 5)) {
        if (!PlayerHelper.isWebWalking()) {
            isExecuting = true;
            var dlx = deathLocation.x;
            var dly = deathLocation.y;
            var dlp = deathLocation.plane;
            Utility.invokeLater(function() {
                PlayerHelper.webWalkTo(new WorldPoint(dlx, dly, dlp));
                isExecuting = false;
            }, Utility.getDelay());
        }
        return;
    }

    // Phase 3: At grave - pick up items from ground
    if (pickUpLoot()) return;

    // Phase 4: Done recovering, return to supply check
    Game.sendGameMessage("Death recovery complete, resuming task.", "SlayerBot");
    deathLocation = null;
    deathDetected = false;
    state = STATES.CHECK_SUPPLIES;
}

        state = STATES.FIGHT_TASK;
        return;

    // Check if we have cannon base
    if (!Game.info.inventory.hasItem(CANNON.BASE, 1)) {
        Game.sendGameMessage("No cannon base in inventory.", "SlayerBot");
        cannonPlaced = false;
        state = STATES.FIGHT_TASK;
        return;
    }

    // Walk to monster area center
    if (!isNear(currentTaskMonster.x, currentTaskMonster.y, 10)) {
        if (!PlayerHelper.isWebWalking()) {
            isExecuting = true;
            var canX = currentTaskMonster.x;
            var canY = currentTaskMonster.y;
            var canP = currentTaskMonster.plane || 0;
            Utility.invokeLater(function() {
                PlayerHelper.webWalkTo(new WorldPoint(canX, canY, canP));
                isExecuting = false;
            }, Utility.getDelay());
        }
        return;

// Multi-phase arrow crafting: mine ore -> smelt -> smith heads -> feathers -> chop -> fletch shafts -> combine
function handleCraftArrows() {
    // Determine best tier
    var tier = getBestArrowCraftingTier(smithingLevel, fletchingLevel, miningLevel);
    if (!tier) {
        Game.sendGameMessage("Cannot craft arrows at current levels (Smith:" + smithingLevel + " Fletch:" + fletchingLevel + " Mine:" + miningLevel + ")", "SlayerBot");
        arrowCraftPhase = "idle";
        state = STATES.CHECK_SUPPLIES;
        return;
    }

    arrowCraftTier = tier;

    // Initialize phase if idle
    if (arrowCraftPhase === "idle") {
        arrowCraftPhase = "check_stock";
        arrowCraftCount = 0;
    }

    switch (arrowCraftPhase) {
        case "check_stock":
            handleArrowCheckStock();
            break;
        case "mine_ore":
            handleArrowMineOre();
            break;
        case "smelt_bars":
            handleArrowSmeltBars();
            break;
        case "smith_heads":
            handleArrowSmithHeads();
            break;
        case "kill_chickens":
            handleArrowKillChickens();
            break;
        case "chop_trees":
            handleArrowChopTrees();
            break;
        case "fletch_shafts":
            handleArrowFletchShafts();
            break;
        case "fletch_arrows":
            handleArrowFletchArrows();
            break;
        case "done":
            handleArrowDone();
            break;
        default:
            arrowCraftPhase = "check_stock";
            break;
    }
}

function handleArrowCheckStock() {
    // Check if we have enough components to skip steps
    var headId = arrowCraftTier.headId;
    var shaftCount = Game.info.inventory.getItemCount(ARROW_SHAFT_ID);
    var featherCount = Game.info.inventory.getItemCount(FEATHER_ID);
    var headCount = Game.info.inventory.getItemCount(headId);

    // Target: 500 arrows minimum
    if (headCount >= 500 && shaftCount >= 500 && featherCount >= 500) {
        arrowCraftPhase = "fletch_arrows";
        return;
    }

    if (headCount < 500) {
        // Need arrowheads: mine ore -> smelt -> smith
        arrowCraftPhase = "mine_ore";
    } else if (featherCount < 500) {
        arrowCraftPhase = "kill_chickens";
    } else if (shaftCount < 500) {
        arrowCraftPhase = "chop_trees";
    } else {
        arrowCraftPhase = "fletch_arrows";
    }
}

// Pipeline: mine essence -> walk to altar -> craft -> bank -> repeat
function handleCraftRunes() {
    if (runecraftPhase === "idle") {
        runecraftPhase = "mine_essence";
        rcEssenceCount = 0;

        // Determine which runes we need
        neededRunes = [];
        var spell = getBestSpell(magicLevel);
        if (spell) {
            var runes = getRunesForSpell(spell.key);
            if (runes) {
                var runeKeys = Object.keys(runes);
                for (var i = 0; i < runeKeys.length; i++) {
                    neededRunes.push({runeId: parseInt(runeKeys[i]), quantity: 1000});
                }
            }
        }
        if (neededRunes.length === 0) {
            // Default to fire runes
            neededRunes.push({runeId: 554, quantity: 1000}); // Fire runes
        }
    }

    switch (runecraftPhase) {
        case "mine_essence":
            handleRCMineEssence();
            break;
        case "walk_to_altar":
            handleRCWalkToAltar();
            break;
        case "craft_runes":
            handleRCCraftRunes();
            break;
        case "bank_runes":
            handleRCBankRunes();
            break;
        case "check_stock":
            handleRCCheckStock();
            break;
        case "done":
            handleRCDone();
            break;
        default:
            runecraftPhase = "mine_essence";
            break;
    }
}

function handleBrewPotions() {
    if (!brewTarget) {
        Game.sendGameMessage("No brew target set, returning to supplies.", "SlayerBot");
        brewPhase = "idle";
        state = STATES.CHECK_SUPPLIES;
        return;
    }

    // Look up the recipe
    var recipe = null;
    var recipeKeys = Object.keys(POTION_RECIPES);
    for (var i = 0; i < recipeKeys.length; i++) {
        var r = dbLookup(POTION_RECIPES, recipeKeys[i]);
        if (r && recipeKeys[i].toLowerCase().indexOf(brewTarget.toLowerCase()) !== -1) {
            recipe = r;
            break;
        }
    }

    if (!recipe) {
        Game.sendGameMessage("Unknown brew target: " + brewTarget, "SlayerBot");
        brewPhase = "idle";
        brewTarget = null;
        state = STATES.CHECK_SUPPLIES;
        return;
    }

    // Check herblore level requirement
    if (herbloreLevel < recipe.level) {
        Game.sendGameMessage("Need herblore level " + recipe.level + " (have " + herbloreLevel + ") for " + brewTarget, "SlayerBot");
        herbloreTargetLevel = recipe.level;
        state = STATES.LEVEL_HERBLORE;
        return;
    }

    if (brewPhase === "idle") {
        brewPhase = "bank_supplies";
        brewCount = 0;
    }

    switch (brewPhase) {
        case "bank_supplies":
            handleBrewBankSupplies(recipe);
            break;
        case "clean_herbs":
            handleBrewCleanHerbs(recipe);
            break;
        case "make_unfinished":
            handleBrewMakeUnfinished(recipe);
            break;
        case "add_secondary":
            handleBrewAddSecondary(recipe);
            break;
        case "bank_potions":
            handleBrewBankPotions();
            break;
        case "done":
            handleBrewDone();
            break;
        default:
            brewPhase = "bank_supplies";
            break;
    }
}

// OnChatMessage -- Intercepts chat messages to detect tasks, kills, deaths
// =============================================================================
function OnChatMessage(chatMsg) {
    if (!chatMsg) return;
    var rawText = "";
    if (typeof chatMsg === "string") {
        rawText = chatMsg;
    } else if (chatMsg.getMessage) {
        rawText = chatMsg.getMessage();
    } else if (chatMsg.message) {
        rawText = chatMsg.message;
    }
    if (!rawText || rawText.length === 0) return;
    var message = rawText;
    var msg = rawText.toLowerCase();

    // Detect task assignment
    for (var p = 0; p < TASK_CHAT_PATTERNS.length; p++) {
        if (msg.indexOf(TASK_CHAT_PATTERNS[p]) !== -1) {
            parseTaskFromMessage(message);
            taskReceived = true;
            return;
        }
    }

    // Detect kill count (e.g., "You have completed your task! You killed 120 monsters.")
    if (msg.indexOf("you have completed your task") !== -1) {
        killsLeft = 0;
        state = STATES.TASK_COMPLETE;
        return;
    }

    // Detect single kill (monster death message or XP drop)
    if (msg.indexOf("your slayer assignment") !== -1 && msg.indexOf("more to go") !== -1) {
        // Parse kills remaining
        var remaining = parseKillsRemaining(msg);
        if (remaining >= 0) {
            killsLeft = remaining;
        }
        return;
    }

    // Detect superior spawn
    if (msg.indexOf("a superior foe has appeared") !== -1) {
        Game.sendGameMessage("Superior monster spawned!", "SlayerBot");
        return;
    }

    // Detect death
    if (msg.indexOf("oh dear, you are dead") !== -1) {
        deathDetected = true;
        return;
    }

    // Detect level up
    if (msg.indexOf("congratulations, you just advanced") !== -1) {
        levelUpDetected = true;
        return;
    }

    // Detect task streaks
    if (msg.indexOf("you've completed") !== -1 && msg.indexOf("in a row") !== -1) {
        // Task streak notification
        Game.sendGameMessage("Task streak tracked!", "SlayerBot");
        return;
    }
}

function parseTaskFromMessage(message) {
    // Parse: "Your new task is to kill 135 Blue Dragons."
    var msg = message;
    var lower = msg.toLowerCase();

    // Find the pattern
    var idx = -1;
    for (var p = 0; p < TASK_CHAT_PATTERNS.length; p++) {
        idx = lower.indexOf(TASK_CHAT_PATTERNS[p]);
        if (idx !== -1) {
            idx = idx + TASK_CHAT_PATTERNS[p].length;
            break;
        }
    }

    if (idx === -1) return;

    // Extract the rest: " 135 Blue Dragons."
    var rest = msg.substring(idx).trim();
    // Remove trailing period
    if (rest.slice(-1) === ".") {
        rest = rest.substring(0, rest.length - 1);
    }

    // Parse number and name
    var parts = rest.split(" ");
    if (parts.length < 2) return;

    var count = parseInt(parts[0]);
    if (isNaN(count)) return;

    var monsterName = parts.slice(1).join(" ");

    killsLeft = count;
    currentTaskName = monsterName;
    totalKills = totalKills + count;

    Game.sendGameMessage("Parsed task: " + count + " x " + monsterName, "SlayerBot");
}

function parseKillsRemaining(msg) {
    // Parse: "...120 more to go."
    var idx = msg.indexOf("more to go");
    if (idx === -1) return -1;

    // Work backwards to find the number
    var numStr = "";
    var i = idx - 1;
    while (i >= 0 && msg.charAt(i) === " ") i--;
    while (i >= 0 && msg.charAt(i) >= "0" && msg.charAt(i) <= "9") {
        numStr = msg.charAt(i) + numStr;
        i--;
    }

    var num = parseInt(numStr);
    if (isNaN(num)) return -1;
    return num;
}

// NPC LOOT APPEARED -- Track kills when NPC dies (more reliable than chat)
// =============================================================================
function OnNPCLootAppeared(npc) {
    if (!currentTaskMonster || !npc) return;

    var npcId = -1;
    try {
        npcId = npc.getId();
    } catch (e) {
        return;
    }

    // Check if this NPC is our task
    var isTask = false;
    if (currentTaskMonster.npcIds) {
        for (var i = 0; i < currentTaskMonster.npcIds.length; i++) {
            if (currentTaskMonster.npcIds[i] === npcId) {
                isTask = true;
                break;
            }
        }
    }

    // Also check superior monster IDs
    if (!isTask) {
        var superiorData = getSuperiorNpcId(currentTaskMonster.name);
        if (superiorData && superiorData === npcId) {
            isTask = true;
        }
    }

    if (isTask) {
        killsLeft--;
        totalKills++;
        if (overlay && overlay.killsLeft) overlay.killsLeft.setValue("Kills Left: " + Math.max(0, killsLeft));
        Game.sendGameMessage("Kill tracked! " + killsLeft + " " + currentTaskMonster.name + " remaining. Total kills: " + totalKills, "SlayerBot");

        if (killsLeft <= 0) {
            Game.sendGameMessage("All kills complete!", "SlayerBot");
            state = STATES.TASK_COMPLETE;
        }
    }
}

// CANNONBALL GATHERING SYSTEM
// Mine iron + coal -> smelt steel bars -> smith cannonballs at furnace
// NOTE: Do NOT bring cannon to single-combat zones!
// ===========================================================================
var gatherCballState = "CHECK";
var gatherCballTarget = 1000;

function handleGatherCannonballs() {
    if (isExecuting) return;

    // Check smithing level
    var smithLvl = Client.getRealSkillLevel(Skill.SMITHING);
    var miningLvl = Client.getRealSkillLevel(Skill.MINING);

    if (smithLvl < CANNONBALL_CRAFT.smithLevel) {
        Game.sendGameMessage("[SlayerBot] Need 35 Smithing to make cannonballs. Current: " + smithLvl, "Bot");
        // Fall back to buying from GE if possible
        gatherCballState = "BUY_GE";
    }

    var cballCount = Game.info.inventory.getItemCount(CANNON.CANNONBALL);
    if (cballCount >= gatherCballTarget) {
        Game.sendGameMessage("[SlayerBot] Have " + cballCount + " cannonballs -- returning to task", "Bot");
        gatherCballState = "CHECK";
        state = STATES.CHECK_SUPPLIES;
        return;
    }

    if (gatherCballState === "CHECK") {
        // Figure out what we need
        var barsInInv = Game.info.inventory.getItemCount(CANNONBALL_CRAFT.steelBar);
        var ironInInv = Game.info.inventory.getItemCount(CANNONBALL_CRAFT.ironOre);
        var coalInInv = Game.info.inventory.getItemCount(CANNONBALL_CRAFT.coal);
        var hasMould = Game.info.inventory.hasItem(CANNONBALL_CRAFT.ammoMould, 1);

        if (barsInInv >= 10) {
            // Have steel bars -- go smith cannonballs
            if (!hasMould) {
                gatherCballState = "GET_MOULD";
            } else {
                gatherCballState = "SMITH_CBALLS";
            }
        } else if (ironInInv > 0 && coalInInv >= (ironInInv * 2)) {
            // Have ores -- go smelt
            gatherCballState = "SMELT_BARS";
        } else {
            // Need to bank for supplies or mine
            gatherCballState = "BANK_SUPPLIES";
        }
        return;
    }

    if (gatherCballState === "BANK_SUPPLIES") {
        if (!Game.info.bank.isOpen()) {
            isExecuting = true;
            Utility.invokeLater(function() {
                Game.interact.bank.openNearest();
                isExecuting = false;
            }, Utility.getDelay());
            return;
        }
        // Deposit inventory
        Game.interact.bank.depositAllInventory();
        // Withdraw ammo mould
        if (!Game.info.inventory.hasItem(CANNONBALL_CRAFT.ammoMould, 1)) {
            withdrawBankItem(CANNONBALL_CRAFT.ammoMould, 1);
        }
        // Try to withdraw steel bars from bank
        withdrawBankItem(CANNONBALL_CRAFT.steelBar, 27);
        isExecuting = true;
        Utility.invokeLater(function() {
            var barsGot = Game.info.inventory.getItemCount(CANNONBALL_CRAFT.steelBar);
            if (barsGot >= 5) {
                Game.interact.bank.close();
                gatherCballState = "SMITH_CBALLS";
            } else {
                // No steel bars in bank -- try to withdraw ores
                withdrawBankItem(CANNONBALL_CRAFT.ironOre, 9);
                withdrawBankItem(CANNONBALL_CRAFT.coal, 18);
                var ironGot = Game.info.inventory.getItemCount(CANNONBALL_CRAFT.ironOre);
                if (ironGot >= 3) {
                    Game.interact.bank.close();
                    gatherCballState = "SMELT_BARS";
                } else {
                    // Nothing in bank -- need to mine
                    Game.interact.bank.close();
                    Game.sendGameMessage("[SlayerBot] No steel bars or ores in bank -- mining iron and coal", "Bot");
                    gatherCballState = "MINE_ORES";
                }
            }
            isExecuting = false;
        }, Utility.getDelay());
        return;
    }

    if (gatherCballState === "MINE_ORES") {
        // Walk to Al Kharid mine (has both iron and coal rocks nearby)
        var minePos = SS_LOCATIONS.AL_KHARID_MINE;
        if (!isNear(minePos.x, minePos.y, 15)) {
            if (!PlayerHelper.isWebWalking()) {
                isExecuting = true;
                Utility.invokeLater(function() {
                    PlayerHelper.webWalkTo(new WorldPoint(minePos.x, minePos.y, minePos.z || 0));
                    isExecuting = false;
                }, Utility.getDelay());
            }
            return;
        }

        var ironCount = Game.info.inventory.getItemCount(CANNONBALL_CRAFT.ironOre);
        var coalCount = Game.info.inventory.getItemCount(CANNONBALL_CRAFT.coal);

        // Need 2 coal per iron for steel bars. Mine ~9 iron + 18 coal per trip
        if (ironCount < 9) {
            // Mine iron
            var ironRock = Game.info.gameObject.getNearest(SS_OBJECTS.IRON_ROCK);
            if (ironRock && PlayerHelper.isPlayerIdle()) {
                isExecuting = true;
                Utility.invokeLater(function() {
                    Game.interact.gameObject.action(ironRock, MenuAction.GAME_OBJECT_FIRST_OPTION);
                    isExecuting = false;
                }, Utility.getDelay());
            }
        } else if (coalCount < (ironCount * 2)) {
            // Mine coal
            var coalRock = Game.info.gameObject.getNearest(SS_OBJECTS.COAL_ROCK);
            if (coalRock && PlayerHelper.isPlayerIdle()) {
                isExecuting = true;
                Utility.invokeLater(function() {
                    Game.interact.gameObject.action(coalRock, MenuAction.GAME_OBJECT_FIRST_OPTION);
                    isExecuting = false;
                }, Utility.getDelay());
            }
        } else {
            // Have enough ores for a trip -- go smelt
            gatherCballState = "SMELT_BARS";
        }

        // Full inventory -- bank or smelt
        if (Game.info.inventory.isFull()) {
            if (ironCount >= 5 && coalCount >= (ironCount * 2)) {
                gatherCballState = "SMELT_BARS";
            } else {
                gatherCballState = "BANK_SUPPLIES";
            }
        }
        return;
    }

    if (gatherCballState === "SMELT_BARS") {
        // Walk to Al Kharid furnace
        var furnacePos = SS_LOCATIONS.AL_KHARID_FURNACE;
        if (!isNear(furnacePos.x, furnacePos.y, 10)) {
            if (!PlayerHelper.isWebWalking()) {
                isExecuting = true;
                Utility.invokeLater(function() {
                    PlayerHelper.webWalkTo(new WorldPoint(furnacePos.x, furnacePos.y, furnacePos.z || 0));
                    isExecuting = false;
                }, Utility.getDelay());
            }
            return;
        }

        // Smelt at furnace
        var ironInv = Game.info.inventory.getItemCount(CANNONBALL_CRAFT.ironOre);
        var coalInv = Game.info.inventory.getItemCount(CANNONBALL_CRAFT.coal);
        if (ironInv > 0 && coalInv >= 2 && PlayerHelper.isPlayerIdle()) {
            var furnace = Game.info.gameObject.getNearest([SS_OBJECTS.FURNACE]);
            if (furnace) {
                isExecuting = true;
                Utility.invokeLater(function() {
                    Game.interact.gameObject.action(furnace, MenuAction.GAME_OBJECT_FIRST_OPTION);
                    Game.sendGameMessage("[SlayerBot] Smelting steel bars for cannonballs", "Bot");
                    isExecuting = false;
                }, Utility.getDelay());
            }
        } else if (ironInv <= 0 || coalInv < 2) {
            // Done smelting -- check steel bar count
            var steelBars = Game.info.inventory.getItemCount(CANNONBALL_CRAFT.steelBar);
            if (steelBars >= 5) {
                gatherCballState = "SMITH_CBALLS";
            } else {
                // Need more ores
                gatherCballState = "BANK_SUPPLIES";
            }
        }
        return;
    }

    if (gatherCballState === "SMITH_CBALLS") {
        // Walk to Al Kharid furnace (cannonballs are smithed at furnace, not anvil)
        var smFurnacePos = SS_LOCATIONS.AL_KHARID_FURNACE;
        if (!isNear(smFurnacePos.x, smFurnacePos.y, 10)) {
            if (!PlayerHelper.isWebWalking()) {
                isExecuting = true;
                Utility.invokeLater(function() {
                    PlayerHelper.webWalkTo(new WorldPoint(smFurnacePos.x, smFurnacePos.y, smFurnacePos.z || 0));
                    isExecuting = false;
                }, Utility.getDelay());
            }
            return;
        }

        // Smith cannonballs: use steel bar on furnace with ammo mould in inventory
        var steelCount = Game.info.inventory.getItemCount(CANNONBALL_CRAFT.steelBar);
        if (steelCount > 0 && Game.info.inventory.hasItem(CANNONBALL_CRAFT.ammoMould, 1) && PlayerHelper.isPlayerIdle()) {
            var smFurnace = Game.info.gameObject.getNearest([SS_OBJECTS.FURNACE]);
            if (smFurnace) {
                isExecuting = true;
                Utility.invokeLater(function() {
                    // Use steel bar on furnace to smith cannonballs
                    Game.interact.inventory.useItemOnObject(CANNONBALL_CRAFT.steelBar, smFurnace);
                    Game.sendGameMessage("[SlayerBot] Smithing cannonballs (" + steelCount + " bars left)", "Bot");
                    isExecuting = false;
                }, Utility.getDelay());
            }
        } else if (steelCount <= 0) {
            // All bars used -- check cannonball count
            var newCballs = Game.info.inventory.getItemCount(CANNON.CANNONBALL);
            Game.sendGameMessage("[SlayerBot] Smithed cannonballs. Total: " + newCballs, "Bot");
            if (newCballs >= gatherCballTarget) {
                gatherCballState = "CHECK";
                state = STATES.CHECK_SUPPLIES;
            } else {
                // Need more -- bank and get more bars/ores
                gatherCballState = "BANK_SUPPLIES";
            }
        }
        return;
    }

    if (gatherCballState === "GET_MOULD") {
        // Buy ammo mould from Nulodion at Dwarf Mines or GE
        // Simplest: try to withdraw from bank first
        if (!Game.info.bank.isOpen()) {
            isExecuting = true;
            Utility.invokeLater(function() {
                Game.interact.bank.openNearest();
                isExecuting = false;
            }, Utility.getDelay());
            return;
        }
        withdrawBankItem(CANNONBALL_CRAFT.ammoMould, 1);
        isExecuting = true;
        Utility.invokeLater(function() {
            Game.interact.bank.close();
            if (Game.info.inventory.hasItem(CANNONBALL_CRAFT.ammoMould, 1)) {
                gatherCballState = "CHECK";
            } else {
                Game.sendGameMessage("[SlayerBot] No ammo mould in bank -- trying GE", "Bot");
                gatherCballState = "BUY_GE";
            }
            isExecuting = false;
        }, Utility.getDelay());
        return;
    }

    if (gatherCballState === "BUY_GE") {
        // Walk to GE and buy cannonballs directly if we have GP
        var gePos = SS_LOCATIONS.GRAND_EXCHANGE;
        if (!isNear(gePos.x, gePos.y, 15)) {
            if (!PlayerHelper.isWebWalking()) {
                isExecuting = true;
                Utility.invokeLater(function() {
                    PlayerHelper.webWalkTo(new WorldPoint(gePos.x, gePos.y, gePos.z || 0));
                    isExecuting = false;
                }, Utility.getDelay());
            }
            return;
        }
        Game.sendGameMessage("[SlayerBot] At GE -- buy cannonballs manually or level Smithing to 35", "Bot");
        gatherCballState = "CHECK";
        state = STATES.CHECK_SUPPLIES;
        return;
    }
}

// OUTSIDE_SCHEDULE -- Handles being outside play hours
// =============================================================================
function handleOutsideSchedule() {
    config.currentStateOverlay.setValue("State: OUTSIDE SCHEDULE");

    // Check if we are back in schedule
    if (config.enableSchedule.getValue()) {
        var schedStart = config.startHour.getValue();
        var schedEnd = config.endHour.getValue();
        if (isWithinSchedule(schedStart, schedEnd)) {
            Game.sendGameMessage("[Schedule] Back within play hours! Resuming.", "Bot");
            config.scheduleOverlay.setValue("Schedule: Active");

            // Reset break timer for the new session
            ticksSinceLastBreak = 0;
            var playMin = config.minPlayMinutes.getValue();
            var playMax = config.maxPlayMinutes.getValue();
            nextBreakAtTick = calculateNextBreakDelay(playMin, playMax);
            sessionStarted = true;

            // Return to previous state
            state = stateBeforeBreak;
            return;
        }
    }

    // If not logged out yet, walk to a bank and log out
    if (!outsideScheduleLogout) {
        // Bank inventory first to save all loot
        if (!Game.info.bank.isOpen()) {
            isExecuting = true;
            Utility.invokeLater(function() {
                Game.interact.bank.openNearest();
                isExecuting = false;
            }, Utility.getDelay());
            return;
        }

        // Deposit everything
        isExecuting = true;
        Utility.invokeLater(function() {
            Game.interact.bank.depositAllInventory();
            isExecuting = false;
        }, Utility.getDelay());

        Utility.invokeLater(function() {
            Game.interact.bank.close();
        }, Utility.getDelay());

        outsideScheduleLogout = true;
        Game.sendGameMessage("[Schedule] Banked items. Waiting for play hours to resume.", "Bot");
        config.scheduleOverlay.setValue("Schedule: WAITING (banked items)");
        return;
    }

    // Already banked -- just idle and wait for schedule to resume
    // Check every ~200 ticks (2 minutes)
    if (tickCounter % 200 === 0) {
        var hour = getCurrentHour();
        config.scheduleOverlay.setValue("Schedule: OFF (hour " + hour + ") waiting...");
    }
}

            break;
